6   package automaattiPokeri.Kayttoliittyma;
7   
8   import automaattiPokeri.KayttoliittymaKuuntelijat.ValikkoNappulaKuuntelija;
9   import java.awt.Color;
10  import java.awt.FlowLayout;
11  import java.awt.Font;
12  import javax.swing.JButton;
13  import javax.swing.JLabel;
14  import javax.swing.JPanel;
15  
16  /**
17   *
18   * @author Juuso
19   */
20  public class Valikko extends JPanel {
21  
22      private MainPanel panel;
23      private ValikkoNappulaKuuntelija kuuntelija;
24      private JLabel nimikentta;
25  
26      /**
27       * Luo uuden valikon
28       *
29       * @param panel Main paneelin ilmentyma.
30       */
31      public Valikko(MainPanel panel) {
32          super();
33          this.setLayout(new FlowLayout());
34          this.panel = panel;
35          luoKuuntelija();
36          createComponents();
37          this.setOpaque(false);
38      }
39  
40      private void createComponents() {
41          lisaaUusiPeliNappula();
42          lisaaLataaPeliNappula();
43          lisaaTopListaNappula();
44          lisaaVaihdaKayttajaNappula();
45          lisaaNimiKentta();
46      }
47  
48      private void lisaaUusiPeliNappula() {
49          JButton nappula = new JButton("Uusi peli");
50          lisaaNappula(nappula);
51      }
52  
53      private void lisaaNappula(JButton nappula) {
54          nappula.addActionListener(kuuntelija);
55          this.add(nappula);
56      }
57  
58      private void lisaaLataaPeliNappula() {
59          JButton nappula = new JButton("Jatka");
60          lisaaNappula(nappula);
61      }
62  
63      private void lisaaTopListaNappula() {
64          JButton nappula = new JButton("Top lista");
65          lisaaNappula(nappula);
66      }
67  
68      private void luoKuuntelija() {
69          kuuntelija = new ValikkoNappulaKuuntelija();
70          kuuntelija.annaMainPanel(panel);
71      }
72  
73      private void lisaaVaihdaKayttajaNappula() {
74          JButton nappula = new JButton("Vaihda Käyttäjä");
75          lisaaNappula(nappula);
76      }
77  
78      /**
79       * Vaihtaa kayttajan nimen valikossa
80       *
81       * @param nimi Uusi nimi
82       */
83      public void vaihdaKayttajaNimi(String nimi) {
84          nimikentta.setText(nimi);
85          this.setSize(this.getPreferredSize().width, this.getPreferredSize().height);
86          this.revalidate();
87          nimikentta.repaint();
88      }
89  
90      private void lisaaNimiKentta() {
91          nimikentta = new JLabel();
92          Font font = new Font(Font.DIALOG, Font.BOLD, 24);
93          nimikentta.setFont(font);
94          nimikentta.setForeground(Color.white);
95          this.add(nimikentta);
96      }
97  
98  }
