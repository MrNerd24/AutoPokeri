3   import automaattiPokeri.Interfaces.KoonMuuttaja;
4   import automaattiPokeri.KayttoliittymaKuuntelijat.NimiKenttaKuuntelija;
5   import java.awt.Color;
6   import java.awt.Component;
7   import java.awt.Dimension;
8   import java.awt.FlowLayout;
9   import java.awt.Font;
10  import javax.swing.JLabel;
11  import javax.swing.JPanel;
12  import javax.swing.JTextField;
13  
14  /**
15   *
16   * @author Juuso
17   */
18  public class PelaajanNimiKysyja extends JPanel implements KoonMuuttaja {
19  
20      private int x;
21      private int y;
22      private MainPanel main;
23  
24      /**
25       * Luo uuden pelaajan nimea kysyvan paneelin.
26       *
27       * @param x vanhemman leveys
28       * @param y vanhemman korkeus
29       * @param main main paneelin ilmentym√§.
30       */
31      public PelaajanNimiKysyja(int x, int y, MainPanel main) {
32          this.x = x;
33          this.y = y;
34          this.main = main;
35          this.setOpaque(false);
36          this.setLayout(new FlowLayout(FlowLayout.CENTER));
37          lisaaTeksti();
38          lisaaTektikentta();
39          muutaKokoa(y, x);
40      }
41  
42      /**
43       * Muuttaa kokoaan vanhemman mukaan
44       *
45       * @param y vanhemman korkeus
46       * @param x vanhemman leveys
47       */
48      @Override
49      public void muutaKokoa(int y, int x) {
50          this.x = x;
51          this.y = y;
52          setBounds((this.x - this.getPreferredSize().width) / 2, (this.y - this.getPreferredSize().height) / 2, this.getPreferredSize().width, this.getPreferredSize().height);
53          this.revalidate();
54          this.repaint();
55      }
56  
57      private void lisaaTeksti() {
58          JLabel teksti = new JLabel("Anna nimesi: ");
59          Font font = new Font(Font.DIALOG, Font.BOLD, 24);
60          teksti.setFont(font);
61          teksti.setForeground(Color.white);
62          teksti.setAlignmentX(Component.CENTER_ALIGNMENT);
63          this.add(teksti);
64      }
65  
66      private void lisaaTektikentta() {
67          JTextField kentta = new JTextField();
68          kentta.setPreferredSize(new Dimension(this.getPreferredSize().width + 20, this.getPreferredSize().height));
69          Font font = new Font(Font.DIALOG, Font.PLAIN, 24);
70          kentta.setFont(font);
71          kentta.setAlignmentX(Component.CENTER_ALIGNMENT);
72          kentta.addActionListener(new NimiKenttaKuuntelija(this));
73          this.add(kentta);
74      }
75  
76      /**
77       * Asettaa pelaajan nimen
78       *
79       * @param nimi pelaajan nimi.
80       */
81      public void setNimi(String nimi) {
82          main.setNimi(nimi);
83      }
84  
85  }
