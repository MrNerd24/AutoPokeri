1   package automaattiPokeri.Kayttoliittyma;
2   
3   import automaattiPokeri.Interfaces.KoonMuuttaja;
4   import automaattiPokeri.Logiikka.TulosLista;
5   import java.util.ArrayList;
6   import javax.swing.JScrollPane;
7   
8   /**
9    *
10   * @author Juuso
11   */
12  public class TuloslistaPanel extends JScrollPane implements KoonMuuttaja {
13  
14      private int x;
15      private int y;
16      private ListaPanel lista;
17      private TulosLista logiikka;
18  
19      /**
20       * Luo uuden tuloslista paneelin
21       *
22       * @param x vanhemman leveys
23       * @param y vanhemmat korkeus
24       * @param lista scrollattava nakyma, jossa itse lista on.
25       */
26      public TuloslistaPanel(int x, int y, ListaPanel lista) {
27          super(lista);
28          this.lista = lista;
29          this.x = x;
30          this.y = y;
31          muutaKokoa(y, x);
32          asetukset();
33          tulostaPelaajatStatsit();
34      }
35  
36      /**
37       * Muuttaa kokoaan vanhemman mukaan
38       *
39       * @param y vanhemman korkeus
40       * @param x vanhemman leveys
41       */
42      @Override
43      public void muutaKokoa(int y, int x) {
44          this.y = y;
45          this.x = x;
46          this.setBounds((x - 800) / 2, 100, 800, y - 150);
47      }
48  
49      private void tulostaPelaajatStatsit() {
50          lista.poistakaikkiLabelit();
51          ArrayList<String> pelaajat = logiikka.getPelaajaStringit();
52          ArrayList<String> rahat = logiikka.getRahaStringit();
53          ArrayList<String> voitot = logiikka.getVoittoStringit();
54          for (int i = 0; i < pelaajat.size(); i++) {
55              String tulos = i + 1 + ". ";
56              tulos += pelaajat.get(i);
57              tulos += ": Suurin rahamäärä: ";
58              tulos += rahat.get(i);
59              tulos += " Suurin voitto: ";
60              tulos += voitot.get(i);
61              lista.lisaaLable(tulos);
62          }
63          lista.paivitaNakyma();
64      }
65  
66      private void asetukset() {
67          this.getViewport().setOpaque(false);
68          this.setOpaque(false);
69          this.setBorder(null);
70          this.setHorizontalScrollBarPolicy(JScrollPane.HORIZONTAL_SCROLLBAR_NEVER);
71          this.setVerticalScrollBarPolicy(JScrollPane.VERTICAL_SCROLLBAR_AS_NEEDED);
72          logiikka = new TulosLista();
73      }
74  
75  }
