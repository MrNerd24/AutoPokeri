11  public class Pelaaja implements Comparable<Pelaaja> {
12  
13      private String kayttajaTunnus;
14      private Double kaynnissaRahaMaara;
15      private Double kaynnissaPanos;
16      private Double suurinRahaMaara;
17      private Double suurinVoitto;
18      private boolean jarjestaSuurimmanVoitonMukaan = false;
19  
20      /**
21       * Luo uuden pelaajan
22       *
23       * @param kayttajaTunnus Pelaajan kayttajatunnus
24       * @param kaynnissaRahaMaara tamanhetkinen rahamaara
25       * @param kaynnissaPanos tamanhetkinen panos
26       * @param suurinRahaMaara suurin rahamaara
27       * @param suurinVoitto suurin voitto.
28       */
29      public Pelaaja(String kayttajaTunnus, String kaynnissaRahaMaara, String kaynnissaPanos, String suurinRahaMaara, String suurinVoitto) {
30          this.kayttajaTunnus = kayttajaTunnus;
31          annaTiedot(kaynnissaRahaMaara, kaynnissaPanos, suurinRahaMaara, suurinVoitto);
32      }
33  
34      private void annaTiedot(String kaynnissaRahaMaara, String kaynnissaPanos, String suurinRahaMaara, String suurinVoitto) throws NumberFormatException {
35          this.kaynnissaRahaMaara = Double.parseDouble(kaynnissaRahaMaara);
36          this.kaynnissaPanos = Double.parseDouble(kaynnissaPanos);
37          this.suurinRahaMaara = Double.parseDouble(suurinRahaMaara);
38          this.suurinVoitto = Double.parseDouble(suurinVoitto);
39      }
40  
41      /**
42       * Alustaa pelaajan ja lukee tilastot tiedostosta.
43       *
44       * @param kayttajaTunnus pelaajan nimi.
45       */
46      public Pelaaja(String kayttajaTunnus) {
47          this.kayttajaTunnus = kayttajaTunnus;
48          TiedostoLukija lukija = new TiedostoLukija(kayttajaTunnus, "pelaajat");
49          if (!lukija.tiedostoOlemassa()) {
50              annaTiedot("5", "1", "0", "0");
51          } else {
52              ArrayList<String> rivit = lukija.getRivit();
53              annaTiedot(rivit.get(1), rivit.get(2), rivit.get(3), rivit.get(4));
54          }
55  
56      }
57

139     /**
140      * Tallentaa pelaajan tiedot tiedostoon.
141      */
142     public void tallennaTiedot() {
143         TiedostoKirjoittaja kirjoittaja = new TiedostoKirjoittaja(kayttajaTunnus, "pelaajat", false);
144         ArrayList<String> rivit = new ArrayList<String>();
145         rivit.add(kayttajaTunnus);
146         rivit.add(kaynnissaRahaMaara.toString());
147         rivit.add(kaynnissaPanos.toString());
148         rivit.add(suurinRahaMaara.toString());
149         rivit.add(suurinVoitto.toString());
150         kirjoittaja.kirjoita(rivit);
151     }

153     @Override
154     public int compareTo(Pelaaja o) {
155         if (jarjestaSuurimmanVoitonMukaan) {
156             if (this.suurinVoitto.compareTo(o.getSuurinVoitto()) != 0) {
157                 return this.suurinVoitto.compareTo(o.getSuurinVoitto());
158             }
159             return this.suurinRahaMaara.compareTo(o.getSuurinRahaMaara());
160         } else {
161             if (this.suurinRahaMaara.compareTo(o.getSuurinRahaMaara()) != 0) {
162                 return this.suurinRahaMaara.compareTo(o.getSuurinRahaMaara());
163             }
164             return this.suurinVoitto.compareTo(o.getSuurinVoitto());
165         }
166     }
